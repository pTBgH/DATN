\section{Từ bảo mật vành đai đến Zero Trust}

\subsection{Hạn chế của mô hình bảo mật dựa trên chu vi}

Trong nhiều thập kỷ, an ninh mạng truyền thống được xây dựng theo mô hình \textbf{phòng thủ dựa trên chu vi}, hay còn gọi là chiến lược ``Lâu đài và Hào nước''. Mô hình này thiết lập ranh giới giữa mạng nội bộ và bên ngoài thông qua tường lửa, VPN và hệ thống IDS/IPS. Đặc trưng quan sát được là cấp quyền truy cập chủ yếu dựa trên vị trí mạng \parencite{kindervag2010build,jericho2004}.

Mặc dù mạng truyền thống vẫn tích hợp một số biện pháp phòng vệ nội bộ (chia VLAN, tường lửa nội bộ, các cơ chế định danh tĩnh ở tầng ứng dụng), mô hình này bộc lộ ba hạn chế khi đối mặt với hạ tầng cloud-native (thiết kế tối ưu cho môi trường đám mây) hiện đại:

\begin{itemize}
    \item \textbf{Sự tin cậy ngầm định:} một khi kẻ tấn công hoặc mối đe doạ nội bộ đã có chỗ đứng bên trong mạng, các hệ thống thường mặc định coi mọi giao tiếp xuất phát từ dải IP nội bộ là an toàn \parencite{nist800207}.
    \item \textbf{Di chuyển ngang:} Khi chu vi đã bị chọc thủng, các biện pháp kiểm soát nội bộ truyền thống chủ yếu dựa trên việc chia mạng theo VLAN/subnet(mạng con) ở quy mô lớn. Kẻ tấn công sau khi chiếm được một máy hợp lệ có thể tái sử dụng danh tính tĩnh (header hoặc token lâu dài) để trinh sát và di chuyển sang các tài nguyên khác trong cùng phân đoạn mạng trước khi bị phát hiện \parencite{kindervag2010build,gilman2017zero}.
    \item \textbf{Sự mờ nhạt của chu vi vật lý:} Môi trường cloud-native, làm việc từ xa và thiết bị cá nhân khiến việc xác định một ``chu vi mạng'' rõ ràng để đặt tường lửa trở nên bất khả thi \parencite{cisa_ztmm2023,nist800207}.
\end{itemize}

\subsection{Định nghĩa và các thuật ngữ}

Đối mặt với các hạn chế trên, khái niệm Zero Trust được John Kindervag đề xuất năm 2010 tại Forrester Research \parencite{kindervag2010build}, kế thừa các nguyên tắc \emph{không còn phụ thuộc chu vi mạng} của Jericho Forum \parencite{jericho2004} và sau đó được chuẩn hoá thành kiến trúc tham chiếu trong \textbf{NIST SP 800-207} \parencite{nist800207}.

Trong các tài liệu hiện nay, ba thuật ngữ ZT, ZTA và ZTE thường bị dùng lẫn lộn. Bảng~\ref{tab:zt_levels} phân biệt chúng theo đúng định nghĩa của NIST SP 800-207 (Mục 2.1) và CISA ZTMM 2.0 \parencite{nist800207,cisa_ztmm2023}.

\begin{table}[H]
\centering
\caption{Phân biệt Zero Trust, ZTA và ZTE}
\label{tab:zt_levels}
\begin{tabularx}{\textwidth}{|l|l|X|}
\hline
\textbf{Thuật ngữ} & \textbf{Viết tắt} & \textbf{Định nghĩa} \\
\hline
Zero Trust & ZT & Tập hợp các \textit{nguyên tắc} an ninh mạng định hướng dịch chuyển phòng thủ từ chu vi mạng tĩnh sang việc tập trung bảo vệ \textit{từng tài sản} (người dùng, thiết bị, dịch vụ). ZT giả định không có sự tin cậy ngầm định cho bất kỳ thành phần nào dựa trên vị trí mạng. \\
\hline
Zero Trust Architecture & ZTA & \textit{Kế hoạch} an ninh mạng của tổ chức tận dụng các nguyên tắc Zero Trust để kiểm soát truy cập tới tài sản, bao gồm: quan hệ giữa các thành phần, thiết kế luồng công việc và chính sách truy cập. ZTA được hiện thực hoá thông qua các công cụ như IdP, PEP và hệ thống quan sát. \\
\hline

\hline
\end{tabularx}
\end{table}

Từ các tenet và mô hình logic được trình bày trong NIST SP 800-207, có thể rút ra hai tư tưởng xuyên suốt mọi triển khai Zero Trust \parencite{nist800207,gilman2017zero}:
\begin{enumerate}
    \item \textbf{Mạng luôn bị coi là đã bị xâm nhập}.
    \item \textbf{Mọi truy cập đều phải được xác thực và uỷ quyền} một cách rõ ràng trên cơ sở \textit{từng yêu cầu}, không được phụ thuộc vào vị trí mạng của chủ thể.
\end{enumerate}

\section{Kiến trúc tham chiếu Zero Trust (NIST SP 800-207)}

\subsection{Bảy nguyên lý}

NIST SP 800-207 định nghĩa 7 nguyên lý nền tảng mà mọi triển khai ZTA phải tuân thủ \parencite{nist800207}:

\begin{table}[H]
\centering
\caption{Bảy nguyên lý Zero Trust}
\label{tab:seven_tenets}
\begin{tabularx}{\textwidth}{|c|p{4cm}|X|}
\hline
\textbf{\#} & \textbf{Nguyên lý} & \textbf{Nội dung} \\
\hline
1 & Mọi thứ đều là tài nguyên & Máy chủ, ứng dụng, API endpoint, CSDL - tất cả đều là tài nguyên cần bảo vệ riêng biệt. \\
\hline
2 & Bảo mật không phụ thuộc vị trí mạng & Không tồn tại vùng mạng được mặc định tin cậy; giao tiếp nội bộ phải được kiểm soát và xác thực với cùng mức độ chặt chẽ như lưu lượng từ bên ngoài. \\
\hline
3 & Cấp quyền theo phiên & Quyền truy cập được cấp tối thiểu, đánh giá lại mỗi phiên, thu hồi ngay sau khi hoàn thành. \\
\hline
4 & Chính sách truy cập động & Quyết định dựa trên danh tính + trạng thái thiết bị + hành vi + thời gian + vị trí. \\
\hline
5 & Giám sát toàn bộ tài sản & Liên tục đánh giá tính toàn vẹn và trạng thái tuân thủ của mọi thiết bị/ứng dụng. \\
\hline
6 & Xác thực liên tục & Không chỉ xác thực lần đầu - xác thực lại khi thực hiện hành động nhạy cảm. \\
\hline
7 & Thu thập tối đa dữ liệu & Thu thập dữ liệu trạng thái về tài sản, hạ tầng mạng và truyền thông để cung cấp đầu vào cho việc ra quyết định và liên tục cải thiện trạng thái bảo mật. \\
\hline
\end{tabularx}
\end{table}

\subsection{Mô hình logic}
\label{subsec:logic_pdp_pep_pip}

NIST SP 800-207 định nghĩa kiến trúc logic ZTA tách biệt hoàn toàn giữa \textbf{control plane (mặt phẳng điều khiển)} và \textbf{data plane (mặt phẳng dữ liệu)} \parencite{nist800207}. NIST SP 1800-35 bổ sung và sử dụng nhất quán bốn vai trò kiến trúc dưới đây, trong đó PE và PA hợp thành \textbf{Policy Decision Point (PDP)} ở control plane, còn PEP thực thi tại data plane và PIP cấp ngữ cảnh cho PDP \parencite{nist_sp_1800_35}.

\begin{figure}[H]
    \centering
    \includegraphics[width=1.1\textwidth]{images/nist_zta_logical.png}
    \caption[Kiến trúc tham chiếu tổng quát của ZTA]{Kiến trúc tham chiếu tổng quát của ZTA \parencite{nist_sp_1800_35}.}
    \label{fig:nist_zta_logical}
\end{figure}

\begin{itemize}
    \item \textbf{Policy Engine (PE) -- Động cơ chính sách:} chịu trách nhiệm đưa ra quyết định cuối cùng về việc cấp, từ chối hoặc thu hồi quyền truy cập tới tài nguyên. PE sử dụng \textit{thuật toán tin cậy} để đánh giá rủi ro của từng yêu cầu.
    \item \textbf{Policy Administrator (PA) -- Quản trị chính sách:} thực thi quyết định của PE bằng cách gửi lệnh tới PEP để thiết lập hoặc ngắt đường truyền giữa chủ thể và tài nguyên.
    \item \textbf{Policy Enforcement Point (PEP) -- Điểm thực thi chính sách:} nằm trực tiếp trên đường truyền dữ liệu, bảo vệ vùng tin cậy chứa tài nguyên, kích hoạt, giám sát và kết thúc các kết nối theo lệnh từ PA.
    \item \textbf{Policy Information Point (PIP) -- Điểm cung cấp thông tin chính sách:} các nguồn thành phần hỗ trợ cấp telemetry và ngữ cảnh (định danh, phân tích bảo mật, tình báo mối đe dọa) cho PDP để PE ra quyết định liên tục.
\end{itemize}

\subsection{Thuật toán tin cậy}
\label{subsec:trust_algorithm}

Để hiểu cách \textbf{Policy Engine ra quyết định} với tập đầu vào do PIP cấp, NIST SP 800-207 đặt tên cho tiến trình đó là \textbf{trust algorithm (TA)} và phân loại các biến thể theo hai trục độc lập \parencite{nist800207}:

\begin{table}[H]
\centering
\caption{Phân loại biến thể \textit{trust algorithm}}
\label{tab:trust_algorithm}
\begin{tabularx}{\textwidth}{|p{3.6cm}|X|}
\hline
\textbf{Trục phân loại} & \textbf{Hai biến thể đối lập} \\
\hline
\textbf{Tiêu chí ra quyết định} \textit{(tiêu chí và.\ điểm số)} & \textit{Dựa trên tiêu chí:} chủ thể phải đạt một tập điều kiện cứng (rule-based, gần với ABAC truyền thống). \textit{Dựa trên điểm số:} PE tính một \textit{confidence/điểm tin cậy} từ tập tín hiệu có trọng số và so với ngưỡng để cấp/từ chối. \\
\hline
	\textbf{Cách dùng ngữ cảnh} \textit{(đơn lẻ và.\ theo ngữ cảnh )} & \textit{Đơn lẻ:} đánh giá từng yêu cầu độc lập, không nhớ lịch sử. \textit{Theo ngữ cảnh:} dùng lịch sử phiên, hành vi gần đây, thay đổi trạng thái bảo mật làm đầu vào -- là điều kiện cần để \textit{đánh giá lại giữa phiên} thay vì chỉ tại lúc mở phiên. \\\hline
\hline
\end{tabularx}
\end{table}

\subsection{Các hướng tiếp cận triển khai ZTA}
\label{subsec:zta_deployment_models}

Khi đã xác định được mô hình logic, tổ chức có thể chọn các \textit{hướng triển khai} khác nhau. NIST SP 800-207 và SP 1800-35 liệt kê bốn hướng tiếp cận phổ biến~\parencite{nist800207,nist_sp_1800_35}:

\begin{itemize}
    \item \textbf{Enhanced Identity Governance (EIG) -- Quản trị Danh tính Nâng cao:} lấy \textit{danh tính} làm trung tâm. Mọi quyết định dựa trên ``chủ thể là ai'' và ``thuộc tính gì''.
    \item \textbf{Phân đoạn Vi mô:} lấy \textit{mạng} làm trung tâm. Chia mạng thành các phân đoạn cực nhỏ bao quanh từng tài nguyên để ngăn chặn di chuyển ngang.
    \item \textbf{Software Defined Perimeter (SDP) -- Chu vi Định nghĩa bằng Phần mềm:} lấy \textit{ứng dụng} làm trung tâm. Tài nguyên bị ẩn hoàn toàn cho đến khi chủ thể được xác thực và uỷ quyền.
    \item \textbf{Secure Access Service Edge (SASE):} lấy \textit{biên} làm trung tâm. Hợp nhất chức năng PEP với các dịch vụ bảo mật được cung cấp qua đám mây tại điểm gần người dùng.
\end{itemize}

Bốn hướng này không loại trừ lẫn nhau. Một thiết kế tối ưu thường là sự giao thoa của EIG, Phân đoạn Vi mô và SDP.

\section{Các khái niệm mở rộng của Zero Trust}

\subsection{Bảo mật thích ứng và CAEP (Continuous Access Evaluation)}
\label{subsec:adaptive_caep}

Khi Policy Engine sử dụng thuật toán tin cậy theo ngữ cảnh, quyết định cấp truy cập không kết thúc tại thời điểm cấp token. Vòng đời truy cập trong ZTA là một vòng đời \emph{đánh giá liên tục}, trong đó PIP tiếp tục đẩy tín hiệu mới (thiết bị bị xâm nhập, hành vi bất thường) tới PDP để \textit{thu hồi quyền truy cập giữa phiên}.

NIST gọi mô hình này là \textbf{Bảo mật thích ứng (adaptive security)} \parencite{nist800207}. Một cách triển khai cụ thể của tầng này là \textbf{Continuous Access Evaluation Profile (CAEP)} \parencite{openid_caep} -- đặc tả do OpenID Foundation duy trì, mô tả mô hình \textit{hướng sự kiện} cho phép IdP phát tín hiệu thay đổi ngữ cảnh tới các PEP để ngắt phiên ngay cả khi token hiện tại chưa hết hạn TTL. Điểm đáng chú ý là vòng lặp này \textit{không đòi hỏi PEP kiểm tra IdP} mà hoạt động theo cơ chế Push.

\subsection{Tiếp cận đa thành phần}
\label{subsec:no_single_product}

NIST SP 800-207 mô tả ZTA bằng các vai trò logic, không gắn vai trò với một sản phẩm thương mại cụ thể~\parencite{nist800207}. Việc gán toàn bộ nền tảng ZTA cho một sản phẩm duy nhất thường dẫn đến \textit{phụ thuộc nhà cung cấp} và không phủ đầy đủ các vai trò bảo mật (ví dụ: một giải pháp mạng tốt có thể không cung cấp khả năng đánh giá trạng thái bảo mật của ứng dụng).

Do đó, xu hướng hiện đại hướng tới tổ hợp \textbf{Đa thành phần} -- chọn công nghệ phù hợp nhất cho từng vai trò logic (PE, PA, PEP, PIP) và liên kết chúng qua các tiêu chuẩn mở (OIDC, SPIFFE, OpenTelemetry). Đây là phương pháp được áp dụng nhất quán trong việc thiết kế kiến trúc hệ thống của đồ án này.

\section{Mô hình Trưởng thành Zero Trust (ZTMM)}

Bên cạnh kiến trúc tham chiếu của NIST, Cơ quan An ninh Mạng và Cơ sở Hạ tầng Hoa Kỳ (CISA) cung cấp một lộ trình thực tiễn dưới dạng \textbf{Mô hình Trưởng thành ZTMM 2.0} \parencite{cisa_ztmm2023}. ZTMM không coi Zero Trust là một trạng thái bật/tắt nhị phân, mà là một cuộc hành trình tiến hóa được cấu thành từ 5 trụ cột, 3 năng lực xuyên suốt và 4 cấp độ trưởng thành.

\begin{figure}[H]
    \centering
    \includegraphics[width=1\textwidth]{images/ZTAPillars.png}
    \caption{Mô hình Trưởng thành Zero Trust của CISA}
    \label{fig:ZTAPillars}
\end{figure}

\subsection{Năm trụ cột chức năng}
\begin{enumerate}
    \item \textbf{Danh tính:} hợp nhất quản lý, thực thi xác thực đa yếu tố và đánh giá rủi ro danh tính liên tục.
    \item \textbf{Thiết bị:} quản lý, kiểm kê và liên tục xác minh tình trạng tuân thủ, sức khỏe của thiết bị truy cập.
    \item \textbf{Mạng lưới:} quản lý luồng lưu lượng động, cô lập tài nguyên qua phân đoạn vi mô và mã hoá toàn diện.
    \item \textbf{Ứng dụng và Workload:} tích hợp bảo mật vào quy trình CI/CD, đánh giá ủy quyền liên tục đối với ứng dụng.
    \item \textbf{Dữ liệu:} phân loại, dán nhãn, mã hoá dữ liệu ở mọi trạng thái và chống thất thoát (DLP).
\end{enumerate}

\subsection{Ba năng lực xuyên suốt}
\begin{enumerate}
    \item \textbf{Tầm nhìn và Phân tích:} Thu thập, quan sát và phân tích các dữ liệu diễn ra trên toàn bộ môi trường để thông báo cho các quyết định chính sách.
    \item \textbf{Tự động hóa và Điều phối:} Tận dụng công cụ tự động để chuyển đổi phản ứng thủ công thành các hành động tự trị, nhất quán và chạy trên quy mô lớn.
    \item \textbf{Quản trị:} Định nghĩa, thực thi và duy trì các chính sách an ninh mạng tập trung, đảm bảo tuân thủ xuyên suốt cả 5 trụ cột.
\end{enumerate}

\subsection{Bốn cấp độ Trưởng thành}
Để đánh giá hệ thống, CISA chia lộ trình thành 4 cấp độ \parencite{cisa_ztmm2023}:
\begin{itemize}
    \item \textbf{Truyền thống:} xác thực tĩnh, phân đoạn mạng vĩ mô, ranh giới tin cậy cố định.
    \item \textbf{Khởi tạo:} bắt đầu tự động hoá, phân quyền dựa trên thuộc tính cơ bản, có khả năng quan sát nội bộ tốt hơn.
    \item \textbf{Nâng cao:} cấu hình tự động, ủy quyền và thực thi chính sách phối hợp chéo giữa các trụ cột, phản hồi dựa trên rủi ro.
    \item \textbf{Tối ưu:} hệ thống tự trị, áp dụng AI/ML để phân tích và cấp quyền truy cập động theo ngưỡng rủi ro thời gian thực.
\end{itemize}
